Какие недостатки вы видите в стиле написания кода этого примера?

function pow(x,n)
{
  let result=1;
  for(let i=0;i<n;i++) {result*=x;}
  return result;
}

let x=prompt("x?",''), n=prompt("n?",'')
if (n<=0)
{
  alert(`Степень ${n} не поддерживается, введите целую степень, большую 0`);
}
else
{
  alert(pow(x,n))
}

function pow(x,n)   //  добавить пробел между параметрами, желательно в теле функции добавить по пустой строке между строками, для лучшей читаемости
{   //  желательно поднять скобку строкой выше
  let result=1; //  добавить пробелы вокруг знака равенства
  for(let i=0;i<n;i++) {result*=x;} //  в параметрах цикла нехватает отступов, тело цикла перенести на отдельную строку
  return result;
}

let x=prompt("x?",''), n=prompt("n?",'')    //  лучше разбить на два отдельных объявления, желательно под строкой сделать отступ
if (n<=0)   //  добавить пробелы в выражении
{   //  желательно поднять скобку строкой выше
  alert(`Степень ${n} не поддерживается, введите целую степень, большую 0`);
}
else
{   //  желательно поднять скобку строкой выше
  alert(pow(x,n))   //  добавить пробелы в аргументах функции pow
}
